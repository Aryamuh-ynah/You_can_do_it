// Rearranging digits to get a multiple of 5......

void solve()
{
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        string s;
        cin >> s;
        for (int i = 0; i < s.size(); ++i) {
            if (s[i] == '0' || s[i] == '5') {
                cout << "Yes\n"; return;
            }
        }
        cout << "No\n";
    }
}
